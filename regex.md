[0-9] -> any digit
[a-zA-Z] -> any alpha

* -> any ammount
. -> wildcard
+ -> one-plus
^ -> not
\t -> tab
\. -> .
\+ -> +
\^ -> ^
\* -> *
( ... ) -> group
\( \) -> ( )
... | ... -> ... or ...
